# Diplomacy / 外交
cheat_add_spy_network = {
    if = {
        limit = {
            has_selected_country = yes
        }
        add_spy_network = {
            target = selected_country
            value = 100
        }
    }
}

# Culture & Religion / 文化与宗教

# Country Modifiers / 国家修正
cheat_toggle_research_speed = {
    if = {
        limit = {
            has_country_modifier = cheat_research_speed_boost
        }
        remove_country_modifier = cheat_research_speed_boost
    }
    else = {
        add_country_modifier_forever = {
            name = cheat_research_speed_boost
        }
    }
}
