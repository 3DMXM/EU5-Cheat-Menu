# 研究部分 - 切换研究速度
cheat_toggle_research_speed_button = {
    scope = player_country

    effect = {
        if = {
            limit = { has_country_modifier = cheat_toggle_research_speed }
            remove_country_modifier = cheat_toggle_research_speed
        }
        else = {
            add_country_modifier = {
                name = cheat_toggle_research_speed
                days = -1
                mode = add
            }
        }
    }
}
